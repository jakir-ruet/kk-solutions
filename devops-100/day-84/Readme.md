### Playbook

```bash
---
- name: Copy index.html to all app servers
  hosts: app_servers
  become: yes
  tasks:
    - name: Ensure /opt/data directory exists
      file:
        path: /opt/data
        state: directory
        mode: '0755'

    - name: Copy index.html to /opt/data on app servers
      copy:
        src: /usr/src/data/index.html
        dest: /opt/data/index.html
        owner: root
        group: root
        mode: '0644'
```
